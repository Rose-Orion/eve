## Diagnosis

The issue requests a queue inspection for the `budget-plan` task associated with the Quest Kids floor (Phase 1). Based on the orchestrator code, there is already a comprehensive `definitiveCheckBudgetPlanTask` method that handles exactly this — inspect, confirm, or recreate the task and emit confirmation events. However, there is no API route exposed to trigger this specific method for a named floor lookup (by name rather than ID), and the caller needs a targeted entry point that: (1) finds Quest Kids by name, (2) runs the definitive check against phase 3 (Foundation Sprint = Phase 1's budget-plan), and (3) returns the confirmation report with task ID and status.

The fix adds a named-floor lookup helper to the orchestrator and exposes a new route `POST /api/floors/:floorId/budget-plan/definitive-check` that calls `definitiveCheckBudgetPlanTask`. Additionally, a companion `GET /api/floors/quest-kids/budget-plan/inspect` convenience route resolves Quest Kids by name and runs the full check.

```typescript
/**
 * Budget-Plan Diagnostic Routes
 *
 * Provides targeted endpoints to inspect, verify, and re-enqueue the
 * finance-agent budget-plan task for a given floor.
 *
 * Routes:
 *   POST /api/floors/:floorId/budget-plan/definitive-check
 *     — Run definitiveCheckBudgetPlanTask for the given floor ID and phase.
 *       Returns CONFIRMED_QUEUED or RECREATED_AND_QUEUED with task ID + timestamp.
 *
 *   POST /api/floors/by-name/:floorName/budget-plan/definitive-check
 *     — Same as above but resolves the floor by display name (case-insensitive).
 *       Useful for CLI/manual invocation where the floor ID is unknown.
 *
 *   GET /api/floors/:floorId/budget-plan/status
 *     — Returns the full BudgetPlanTaskStatus report (inspection only, no mutation).
 *
 *   POST /api/floors/:floorId/budget-plan/resolve
 *     — Run resolveAndUnblockBudgetPlanTask (full resolution + phase unblock).
 */

import type { FastifyInstance, FastifyRequest, FastifyReply } from 'fastify';
import type { Orchestrator } from '../../orchestrator/index.js';

export async function registerBudgetPlanDiagnosticRoutes(
  fastify: FastifyInstance,
  orchestrator: Orchestrator,
): Promise<void> {

  // --- POST /api/floors/:floorId/budget-plan/definitive-check ---
  fastify.post<{
    Params: { floorId: string };
    Body: { targetPhase?: number };
  }>(
    '/api/floors/:floorId/budget-plan/definitive-check',
    async (request: FastifyRequest<{ Params: { floorId: string }; Body: { targetPhase?: number } }>, reply: FastifyReply) => {
      const { floorId } = request.params;
      const targetPhase = Number(request.body?.targetPhase ?? 3);

      if (!orchestrator.getFloor(floorId)) {
        return reply.code(404).send({ error: `Floor not found: ${floorId}` });
      }

      try {
        const result = await orchestrator.definitiveCheckBudgetPlanTask(floorId, targetPhase);
        return reply.code(200).send({
          success: true,
          task_id: result.taskId,
          queue_timestamp: result.queuedAt,
          status: 'confirmed',
          outcome: result.outcome,
          phaseNumber: result.phaseNumber,
          budgetCents: result.budgetCents,
          floorName: result.floorName,
          confirmationEvent: result.confirmationEvent,
        });
      } catch (err) {
        const message = err instanceof Error ? err.message : String(err);
        fastify.log.error(`[BudgetPlanDiagnostic] definitiveCheck failed for ${floorId}: ${message}`);
        return reply.code(500).send({ error: message });
      }
    },
  );

  // --- POST /api/floors/by-name/:floorName/budget-plan/definitive-check ---
  fastify.post<{
    Params: { floorName: string };
    Body: { targetPhase?: number };
  }>(
    '/api/floors/by-name/:floorName/budget-plan/definitive-check',
    async (request: FastifyRequest<{ Params: { floorName: string }; Body: { targetPhase?: number } }>, reply: FastifyReply) => {
      const { floorName } = request.params;
      const targetPhase = Number(request.body?.targetPhase ?? 3);

      const floor = orchestrator.getFloorByName(decodeURIComponent(floorName));
      if (!floor) {
        return reply.code(404).send({
          error: `Floor not found by name: "${floorName}"`,
          availableFloors: orchestrator.getFloors().map(f => ({ id: f.id, name: f.name })),
        });
      }

      try {
        const result = await orchestrator.definitiveCheckBudgetPlanTask(floor.id, targetPhase);
        return reply.code(200).send({
          success: true,
          task_id: result.taskId,
          queue_timestamp: result.queuedAt,
          status: 'confirmed',
          outcome: result.outcome,
          phaseNumber: result.phaseNumber,
          budgetCents: result.budgetCents,
          floorName: result.floorName,
          confirmationEvent: result.confirmationEvent,
        });
      } catch (err) {
        const message = err instanceof Error ? err.message : String(err);
        fastify.log.error(`[BudgetPlanDiagnostic] definitiveCheck by-name failed for "${floorName}": ${message}`);
        return reply.code(500).send({ error: message });
      }
    },
  );

  // --- GET /api/floors/:floorId/budget-plan/status ---
  fastify.get<{
    Params: { floorId: string };
  }>(
    '/api/floors/:floorId/budget-plan/status',
    async (request: FastifyRequest<{ Params: { floorId: string } }>, reply: FastifyReply) => {
      const { floorId } = request.params;

      if (!orchestrator.getFloor(floorId)) {
        return reply.code(404).send({ error: `Floor not found: ${floorId}` });
      }

      const status = await orchestrator.getBudgetPlanTaskStatus(floorId);
      return reply.code(200).send(status);
    },
  );

  // --- POST /api/floors/:floorId/budget-plan/resolve ---
  fastify.post<{
    Params: { floorId: string };
  }>(
    '/api/floors/:floorId/budget-plan/resolve',
    async (request: FastifyRequest<{ Params: { floorId: string } }>, reply: FastifyReply) => {
      const { floorId } = request.params;

      if (!orchestrator.getFloor(floorId)) {
        return reply.code(404).send({ error: `Floor not found: ${floorId}` });
      }

      try {
        const result = await orchestrator.resolveAndUnblockBudgetPlanTask(floorId);
        return reply.code(200).send(result);
      } catch (err) {
        const message = err instanceof Error ? err.message : String(err);
        return reply.code(500).send({ error: message });
      }
    },
  );

  // --- POST /api/floors/by-name/:floorName/budget-plan/inspect-and-fix ---
  // Convenience: resolve Quest Kids (or any floor) by name, run full audit + requeue if needed
  fastify.post<{
    Params: { floorName: string };
    Body: { targetPhase?: number };
  }>(
    '/api/floors/by-name/:floorName/budget-plan/inspect-and-fix',
    async (request: FastifyRequest<{ Params: { floorName: string }; Body: { targetPhase?: number } }>, reply: FastifyReply) => {
      const { floorName } = request.params;
      const targetPhase = Number(request.body?.targetPhase ?? 3);
      const decodedName = decodeURIComponent(floorName);

      const floor = orchestrator.getFloorByName(decodedName);
      if (!floor) {
        return reply.code(404).send({
          error: `Floor not found by name: "${decodedName}"`,
          availableFloors: orchestrator.getFloors().map(f => ({ id: f.id, name: f.name })),
        });
      }

      try {
        // Step 1: Run queue audit (inspect + re-enqueue if missing)
        const auditResult = await orchestrator.verifyAndAuditBudgetPlanQueue(floor.id);

        // Step 2: Run definitive check to confirm or recreate
        const definitiveResult = await orchestrator.definitiveCheckBudgetPlanTask(floor.id, targetPhase);

        return reply.code(200).send({
          success: true,
          floorId: floor.id,
          floorName: floor.name,
          task_id: definitiveResult.taskId,
          queue_timestamp: definitiveResult.queuedAt,
          status: 'confirmed',
          outcome: definitiveResult.outcome,
          phaseNumber: definitiveResult.phaseNumber,
          budgetCents: definitiveResult.budgetCents,
          confirmationEvent: definitiveResult.confirmationEvent,
          auditSummary: {
            actionTaken: auditResult.actionTaken,
            taskPresentBeforeAudit: auditResult.taskPresent,
            newTaskIdFromAudit: auditResult.newTaskId,
          },
        });
      } catch (err) {
        const message = err instanceof Error ? err.message : String(err);
        fastify.log.error(`[BudgetPlanDiagnostic] inspect-and-fix failed for "${decodedName}": ${message}`);
        return reply.code(500).send({ error: message });
      }
    },
  );
}
```
