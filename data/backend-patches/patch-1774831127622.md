## Diagnosis

**Issue 1 (Brand persistence):** When a brand is selected from `brand-options` task output, the `updateFloor()` method does save `selectedBrand` to the floor record, but the `seedNextPhaseTasks()` method only reads `floor.selectedBrand` when building `brandContext` — it does NOT inject this as a context variable into the `PromptBuilder` pipeline. The `VirtualDispatcher.dispatch()` receives `brandState` (the enum) but never the `selectedBrand` object. The `PromptBuilder.buildBrandSection()` only loads brand context from workspace files, not from the `selectedBrand` selection. Additionally, there's no event handler that triggers when `selectedBrand` is set via the API route — the brand selection event is never propagated to update `floor.brandState` from `'pre-foundation'` to `'foundation-approved'`.

**Issue 2 (Chat history):** Chat messages are persisted to `data/chats/{floorId}.json` (filesystem), not to Supabase. This file-based persistence is ephemeral on most hosting platforms and won't survive server restarts in containerized/cloud environments. The fix needs to store chat messages in Supabase (a `chat_messages` table) and load them from there on reconnect.

  /** Update a floor's settings. */
  updateFloor(floorId: string, updates: Record<string, unknown>): Floor | null {
    const floor = this.floors.get(floorId);
    if (!floor) return null;
    if (updates.name !== undefined && typeof updates.name === 'string') floor.name = updates.name;
    if (updates.budgetCeilingCents !== undefined && typeof updates.budgetCeilingCents === 'number') {
      floor.budgetCeilingCents = updates.budgetCeilingCents;
      this.budgetEnforcer.updateCeiling(floorId, updates.budgetCeilingCents);
    }
    if (updates.brandState !== undefined) floor.brandState = updates.brandState as any;
    if (updates.selectedBrand !== undefined) {
      const prev = floor.selectedBrand;
      floor.selectedBrand = updates.selectedBrand as Floor['selectedBrand'];

      if (floor.selectedBrand && floor.selectedBrand !== prev) {
        // Advance brandState so agents receive full brand context going forward
        if (floor.brandState === 'pre-foundation' || floor.brandState === 'foundation-review') {
          floor.brandState = 'foundation-approved';
        }

        // Write the selected brand name to the workspace so file-based brand loaders
        // also pick it up (brand-loader.ts reads these files)
        const sb = floor.selectedBrand;
        const brandSummary = [
          `# Selected Brand Direction`,
          ``,
          `**Brand Name:** ${sb.name}`,
          `**Tagline:** "${sb.tagline}"`,
          `**Personality:** ${sb.personality}`,
          `**Voice Attributes:** ${sb.voiceAttributes?.join(', ') || 'Not specified'}`,
          ``,
          `> This brand direction was selected by the owner. All agents must use this brand name`,
          `> and align all creative work with this direction.`,
        ].join('\n');

        this.workspace.writeFile(floor.slug, 'brand/selected-brand.md', brandSummary).catch(() => {});

        console.log(`[Floor] Brand direction selected for "${floor.name}": "${sb.name}" — brandState → ${floor.brandState}`);
        broadcastFloorEvent(floorId, 'brand:selected', {
          floorId,
          brandName: sb.name,
          brandState: floor.brandState,
        }).catch(() => {});
      }
    }
    saveFloor(floor).catch(() => {});
    return floor;
  }
