  // --- Analytics Agent Readiness Verification (Public API) ---

  /**
   * Investigate the analytics-agent readiness verification task for a floor/phase
   * (specifically the Phase 9 prerequisite for SideQuest).
   *
   * Steps:
   * 1. SEARCH — scan task queue, event logs (all tasks in memory), and DB for a task
   *    matching 'analytics-agent readiness verification' or similar task types on the
   *    target phase.
   * 2. DETERMINE STATUS — pending / in-progress / completed / failed / never-started.
   * 3. CREATE IF MISSING — if no record found, create and queue the verification task now.
   * 4. EMIT CLEARANCE EVENT — upon confirmed readiness (task completed or newly queued),
   *    emit 'floor:phase-clearance' so floor-manager can re-queue the three pending
   *    analytics tasks (performance-review, ad-optimization, growth-report).
   * 5. AUDIT — persist findings to review-log.jsonl.
   *
   * Returns a structured report including task ID, status, last-state-change timestamp,
   * and whether the clearance event was successfully emitted.
   */
  async investigateAnalyticsAgentReadiness(
    floorId: string,
    phaseNumber = 9,
  ): Promise<{
    floorId: string;
    floorName: string;
    phaseNumber: number;
    investigatedAt: string;
    verificationTaskId: string | null;
    verificationTaskStatus: 'pending' | 'in-progress' | 'completed' | 'failed' | 'never-started';
    lastStateChangeAt: string | null;
    taskFound: boolean;
    taskCreatedNow: boolean;
    clearanceEventEmitted: boolean;
    pendingAnalyticsTasks: Array<{ taskType: string; taskId: string; status: string }>;
    findings: string[];
    recommendation: string;
  }> {
    const investigatedAt = new Date().toISOString();
    const floor = this.floors.get(floorId);
    const floorName = floor?.name ?? 'Unknown';
    const findings: string[] = [];

    // --- STEP 1: SEARCH for analytics-agent readiness verification task ---
    // Match by taskType or description containing readiness/verification markers.
    const READINESS_TASK_TYPES = [
      'analytics-agent-readiness-verification',
      'analytics-agent-readiness',
      'analytics-readiness-verification',
      'analytics-readiness',
      'agent-readiness-verification',
    ];
    const READINESS_DESCRIPTION_MARKERS = [
      'readiness verification',
      'readiness check',
      'analytics-agent readiness',
      'analytics agent readiness',
      'phase 9 prerequisite',
    ];

    const allFloorTasks = this.taskManager.getFloorTasks(floorId);

    // Search in-memory tasks
    let verificationTask = allFloorTasks.find(t =>
      t.phaseNumber === phaseNumber &&
      t.assignedAgent === 'analytics-agent' &&
      (
        READINESS_TASK_TYPES.includes(t.taskType) ||
        READINESS_DESCRIPTION_MARKERS.some(
          m => t.description.toLowerCase().includes(m) || t.taskType.toLowerCase().includes('readiness'),
        )
      ),
    ) ?? null;

    // Also check DB if not found in memory
    let dbVerificationTask: { id: string; status: string; task_type: string; created_at: string; updated_at: string | null } | null = null;
    if (!verificationTask) {
      try {
        const { getSupabase } = await import('../integrations/supabase.js');
        const sb = getSupabase();
        if (sb) {
          const { data } = await sb
            .from('tasks')
            .select('id, status, task_type, created_at, updated_at')
            .eq('floor_id', floorId)
            .eq('phase_number', phaseNumber)
            .eq('assigned_agent', 'analytics-agent')
            .order('created_at', { ascending: false });

          if (data) {
            const dbMatch = (data as Array<{ id: string; status: string; task_type: string; created_at: string; updated_at: string | null }>)
              .find(r =>
                READINESS_TASK_TYPES.includes(r.task_type) ||
                r.task_type.toLowerCase().includes('readiness'),
              ) ?? null;
            if (dbMatch) {
              dbVerificationTask = dbMatch;
              findings.push(
                `FINDING: Analytics-agent readiness verification task found in DATABASE only ` +
                `(not in orchestrator memory): id=${dbMatch.id.slice(0, 8)}, ` +
                `status=${dbMatch.status}, taskType=${dbMatch.task_type}.`,
              );
            }
          }
        }
      } catch (dbErr) {
        findings.push(`DB query for verification task failed: ${(dbErr as Error).message}`);
      }
    }

    // --- STEP 2: DETERMINE STATUS ---
    const ACTIVE_STATUSES = ['queued', 'created', 'dispatched', 'working'];
    let verificationTaskId: string | null = null;
    let verificationTaskStatus: 'pending' | 'in-progress' | 'completed' | 'failed' | 'never-started' = 'never-started';
    let lastStateChangeAt: string | null = null;
    let taskFound = false;
    let taskCreatedNow = false;

    if (verificationTask) {
      taskFound = true;
      verificationTaskId = verificationTask.id;
      lastStateChangeAt = verificationTask.completedAt?.toISOString()
        ?? verificationTask.dispatchedAt?.toISOString()
        ?? verificationTask.createdAt?.toISOString()
        ?? null;

      if (verificationTask.status === 'completed') {
        verificationTaskStatus = 'completed';
      } else if (verificationTask.status === 'escalated' || verificationTask.status === 'failed') {
        verificationTaskStatus = 'failed';
      } else if (verificationTask.status === 'dispatched' || verificationTask.status === 'working') {
        verificationTaskStatus = 'in-progress';
      } else {
        verificationTaskStatus = 'pending';
      }

      findings.push(
        `FINDING: Analytics-agent readiness verification task FOUND in orchestrator memory: ` +
        `id=${verificationTask.id.slice(0, 8)}, taskType=${verificationTask.taskType}, ` +
        `status=${verificationTask.status}, phase=${verificationTask.phaseNumber}, ` +
        `attempts=${verificationTask.attempts}/${verificationTask.maxAttempts}, ` +
        `createdAt=${verificationTask.createdAt.toISOString()}.`,
      );
    } else if (dbVerificationTask) {
      taskFound = true;
      verificationTaskId = dbVerificationTask.id;
      lastStateChangeAt = dbVerificationTask.updated_at ?? dbVerificationTask.created_at;

      const dbStatus = dbVerificationTask.status;
      if (dbStatus === 'completed') verificationTaskStatus = 'completed';
      else if (dbStatus === 'escalated' || dbStatus === 'failed') verificationTaskStatus = 'failed';
      else if (dbStatus === 'dispatched' || dbStatus === 'working') verificationTaskStatus = 'in-progress';
      else verificationTaskStatus = 'pending';

      findings.push(
        `FINDING: Verification task found in DB (not memory): ` +
        `status=${dbVerificationTask.status}, taskType=${dbVerificationTask.task_type}.`,
      );
    } else {
      verificationTaskStatus = 'never-started';
      findings.push(
        `FINDING: NO analytics-agent readiness verification task found in orchestrator memory ` +
        `OR database for floor "${floorName}" (phase ${phaseNumber}). Task was NEVER started.`,
      );
    }

    // --- STEP 3: CREATE IF MISSING ---
    // Create the verification task if never started or permanently failed
    const shouldCreate = !taskFound || verificationTaskStatus === 'failed' || verificationTaskStatus === 'never-started';

    if (shouldCreate && floor) {
      const verificationPrompt =
        `Perform analytics-agent readiness verification for "${floor.name}" Phase ${phaseNumber}.\n\n` +
        `This is a prerequisite check before the three Phase ${phaseNumber} analytics tasks ` +
        `(performance-review, ad-optimization, growth-report) are dispatched.\n\n` +
        `Verify:\n` +
        `1. Analytics tracking is configured (UTM parameters, conversion pixels)\n` +
        `2. Prior phase deliverables are available (analytics-setup, conversion-tracking from phases 7-8)\n` +
        `3. Performance baseline data exists or can be established\n` +
        `4. Ad campaign data is accessible for optimization analysis\n\n` +
        `Output: A readiness checklist with PASS/FAIL for each item and a final READY or ` +
        `NOT READY verdict. If READY, state "ANALYTICS AGENT READY FOR PHASE ${phaseNumber}".`;

      const newVerificationTask = this.taskManager.create({
        floorId,
        phaseNumber,
        assignedAgent: 'analytics-agent',
        modelTier: this.modelRouter.getModelTier('analytics-agent', 'routine'),
        taskType: 'analytics-agent-readiness-verification',
        description: `Analytics-agent readiness verification — Phase ${phaseNumber} prerequisite for "${floor.name}"`,
        prompt: verificationPrompt,
        priority: 'high',
        outputFiles: [`deliverables/analytics-readiness-phase${phaseNumber}.md`],
      });

      this.dependencyGraph.addTask(newVerificationTask.id, newVerificationTask.dependsOn);
      verificationTaskId = newVerificationTask.id;
      verificationTaskStatus = 'pending';
      lastStateChangeAt = new Date().toISOString();
      taskCreatedNow = true;
      taskFound = true;

      findings.push(
        `ACTION: Readiness verification task CREATED and QUEUED: ` +
        `id=${newVerificationTask.id.slice(0, 8)}, priority=high, phase=${phaseNumber}. ` +
        `Will dispatch in next processing cycle.`,
      );

      await broadcastFloorEvent(floorId, 'task:created', {
        taskId: newVerificationTask.id,
        floorId,
        taskType: 'analytics-agent-readiness-verification',
      }).catch(() => {});

      // Update local reference for downstream steps
      verificationTask = newVerificationTask;
    }

    // --- STEP 4: IDENTIFY PENDING ANALYTICS TASKS ---
    const ANALYTICS_TASK_TYPES = ['performance-review', 'ad-optimization', 'growth-report'];
    const pendingAnalyticsTasks = allFloorTasks
      .filter(t =>
        t.phaseNumber === phaseNumber &&
        ANALYTICS_TASK_TYPES.includes(t.taskType) &&
        t.status !== 'completed',
      )
      .map(t => ({ taskType: t.taskType, taskId: t.id, status: t.status }));

    findings.push(
      `PENDING ANALYTICS TASKS for phase ${phaseNumber}: ` +
      (pendingAnalyticsTasks.length > 0
        ? pendingAnalyticsTasks.map(t => `${t.taskType} (${t.taskId.slice(0, 8)}, ${t.status})`).join(', ')
        : 'none found in memory — tasks may need to be seeded'),
    );

    // If none found, check if they need to be seeded
    if (pendingAnalyticsTasks.length === 0 && floor) {
      findings.push(
        `NOTE: performance-review, ad-optimization, growth-report tasks not found in memory ` +
        `for phase ${phaseNumber}. They will be seeded when the phase is activated via ` +
        `seedNextPhaseTasks after verification passes.`,
      );
    }

    // --- STEP 5: EMIT CLEARANCE EVENT ---
    // Clearance is appropriate when:
    //   (a) verification task is completed, OR
    //   (b) verification task was just created/queued (pending clearance conditional on completion)
    // We emit the event in both cases so floor-manager can act:
    //   - For 'completed': full clearance
    //   - For 'pending'/'in-progress': conditional clearance (monitor for completion)
    let clearanceEventEmitted = false;
    const isReadyForClearance =
      verificationTaskStatus === 'completed' ||
      verificationTaskStatus === 'pending' ||
      verificationTaskStatus === 'in-progress';

    if (isReadyForClearance && verificationTaskId) {
      const clearancePayload = {
        floorId,
        floorName,
        phaseNumber,
        verificationTaskId,
        verificationTaskStatus,
        clearanceType: verificationTaskStatus === 'completed'
          ? 'full-clearance'
          : 'conditional-clearance',
        pendingAnalyticsTaskTypes: ANALYTICS_TASK_TYPES,
        pendingAnalyticsTasks,
        message:
          verificationTaskStatus === 'completed'
            ? `Analytics-agent readiness CONFIRMED for "${floorName}" Phase ${phaseNumber}. ` +
              `Floor-manager cleared to re-queue: performance-review, ad-optimization, growth-report.`
            : `Analytics-agent readiness verification task ${verificationTaskId.slice(0, 8)} is ${verificationTaskStatus}. ` +
              `Conditional clearance issued — floor-manager should re-queue analytics tasks ` +
              `once verification completes. Tasks: performance-review, ad-optimization, growth-report.`,
        trigger: 'investigateAnalyticsAgentReadiness',
        timestamp: new Date().toISOString(),
      };

      try {
        await broadcastFloorEvent(floorId, 'floor:phase-clearance', clearancePayload);
        clearanceEventEmitted = true;
        findings.push(
          `ACTION: 'floor:phase-clearance' event emitted (${clearancePayload.clearanceType}). ` +
          `Floor-manager can now re-queue: performance-review, ad-optimization, growth-report.`,
        );

        // Also emit on internal EventBus so any in-process listeners receive it
        this.eventBus.emit('floor:phase-started', { floorId, phase: phaseNumber });

        // Submit to floor-manager feedback chain
        if (floor) {
          this.improvementEngine.submitAgentFeedback(
            floorId,
            'floor-manager' as any,
            clearancePayload.message,
            { name: floorName, phase: floor.currentPhase || phaseNumber, status: floor.status, goal: floor.goal },
          ).catch(() => {});
        }
      } catch (emitErr) {
        findings.push(
          `ERROR: Failed to emit floor:phase-clearance event: ${(emitErr as Error).message}`,
        );
      }
    } else if (verificationTaskStatus === 'failed') {
      findings.push(
        `WARNING: Clearance event NOT emitted — verification task is in failed/escalated state. ` +
        `A new verification task has been created (see above). ` +
        `Clearance will be emitted once the new task completes.`,
      );
    }

    // --- STEP 6: SEND NOTIFICATION if task was created ---
    if (taskCreatedNow) {
      sendNotification({
        title: 'Analytics Readiness Verification Created',
        body: `Phase ${phaseNumber} prerequisite task created for "${floorName}": ` +
          `analytics-agent readiness verification (id: ${verificationTaskId?.slice(0, 8)}). ` +
          `Conditional clearance issued to floor-manager for 3 analytics tasks.`,
        floorId,
        type: 'info',
      });
    }

    // --- STEP 7: AUDIT RECORD ---
    await this.writeSystemReviewLog({
      event: 'analytics_agent_readiness_investigation',
      floorId,
      floorName,
      phaseNumber,
      investigatedAt,
      verificationTaskId,
      verificationTaskStatus,
      lastStateChangeAt,
      taskFound,
      taskCreatedNow,
      clearanceEventEmitted,
      pendingAnalyticsTasks,
      findings,
    });

    console.log(
      `[AnalyticsReadiness] "${floorName}" phase ${phaseNumber}: ` +
      `taskFound=${taskFound}, status=${verificationTaskStatus}, ` +
      `taskCreatedNow=${taskCreatedNow}, clearanceEmitted=${clearanceEventEmitted}`,
    );

    // Build recommendation
    let recommendation: string;
    if (verificationTaskStatus === 'completed') {
      recommendation =
        `Analytics-agent readiness CONFIRMED for phase ${phaseNumber}. ` +
        `Full clearance emitted — floor-manager should re-queue the 3 analytics tasks immediately.`;
    } else if (taskCreatedNow) {
      recommendation =
        `Verification task was NEVER STARTED — created and queued now ` +
        `(id: ${verificationTaskId?.slice(0, 8)}). ` +
        `Conditional clearance issued. Monitor for completion, then re-queue ` +
        `performance-review, ad-optimization, growth-report.`;
    } else if (verificationTaskStatus === 'in-progress') {
      recommendation =
        `Verification task is IN PROGRESS (${verificationTask?.status ?? 'dispatched/working'}). ` +
        `Conditional clearance emitted. Await completion before dispatching analytics tasks.`;
    } else if (verificationTaskStatus === 'pending') {
      recommendation =
        `Verification task is PENDING (queued, not yet dispatched). ` +
        `Conditional clearance emitted. Will dispatch in next processing cycle.`;
    } else if (verificationTaskStatus === 'failed') {
      recommendation =
        `Prior verification task FAILED. New task created and queued. ` +
        `Monitor next dispatch cycle.`;
    } else {
      recommendation = `Investigation complete. Review findings for next steps.`;
    }

    return {
      floorId,
      floorName,
      phaseNumber,
      investigatedAt,
      verificationTaskId,
      verificationTaskStatus,
      lastStateChangeAt,
      taskFound,
      taskCreatedNow,
      clearanceEventEmitted,
      pendingAnalyticsTasks,
      findings,
      recommendation,
    };
  }

  // --- Phase Task Queue Investigation (Public API) ---
