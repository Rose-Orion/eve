## Diagnosis

The SideQuest Phase 8 stall occurs because `seedNextPhaseTasks()` creates `ad-creative-production` and `conversion-tracking` tasks for Phase 8 after the budget-plan task completes Phase 3, but there is **no task-completion event handler** that fires `seedNextPhaseTasks` for Phase 8 specifically after the Phase 7 gate is approved. The `checkPhaseCompletion` handler in `setupEventHandlers` only calls `completePhase()` on the PhaseManager — which for Phase 8 (a gate phase, `requiresGate: true`) moves it to `gate-waiting` instead of auto-advancing. The downstream Phase 8 execution tasks (`ad-creative-production`, `conversion-tracking`) are **only seeded when `approvePhaseGate(7)` is called**, which triggers `seedNextPhaseTasks(floorId, floor, 7)`. If Phase 8 tasks are missing after budget-plan completed, the issue is that Phase 7's gate was never approved or `seedNextPhaseTasks` for phase 8 was never called.

The fix has two parts: (1) a new public method `enqueueMissingPhase8Tasks` that checks for missing Phase 8 tasks and re-seeds them, and (2) a route-level endpoint to trigger it for SideQuest. Since the issue states "budget-plan completed but no downstream execution tasks were queued," the most targeted fix is to add a repair method that can be called via the API to re-enqueue the missing Phase 8 tasks without re-running the gate approval flow.

  // --- Phase Task Re-Enqueue (Public API) ---

  /**
   * Inspect a floor's task queue for a given phase and re-enqueue any missing
   * execution tasks. Idempotent — if tasks already exist in any state, returns
   * immediately without creating duplicates.
   *
   * Used to recover Phase 8 (Ad Campaign Activation) after the budget-plan task
   * completed but downstream execution tasks were never created.
   *
   * Steps:
   * 1. Verify floor exists and phase number is valid (4–10).
   * 2. Check in-memory task map AND Supabase for existing phase tasks.
   * 3. If tasks exist → return 'already_present' (idempotent).
   * 4. If missing → call seedNextPhaseTasks for (phase - 1) → phase.
   * 5. Verify newly created tasks are in 'queued' state.
   * 6. Broadcast task:created events for each new task.
   * 7. Audit log the action.
   */
  async enqueueMissingPhaseTasks(
    floorId: string,
    phaseNumber: number,
  ): Promise<{
    floorId: string;
    floorName: string;
    phaseNumber: number;
    outcome: 'already_present' | 'seeded' | 'error';
    tasksCreated: number;
    taskSummary: Array<{ taskId: string; taskType: string; assignedAgent: string; status: string }>;
    error?: string;
    enqueuedAt: string;
  }> {
    const enqueuedAt = new Date().toISOString();
    const floor = this.floors.get(floorId);

    if (!floor) {
      return {
        floorId, floorName: 'Unknown', phaseNumber,
        outcome: 'error', tasksCreated: 0, taskSummary: [],
        error: `Floor not found: ${floorId}`,
        enqueuedAt,
      };
    }

    if (phaseNumber < 4 || phaseNumber > 10) {
      return {
        floorId, floorName: floor.name, phaseNumber,
        outcome: 'error', tasksCreated: 0, taskSummary: [],
        error: `Phase ${phaseNumber} is not a valid seedable phase (4–10).`,
        enqueuedAt,
      };
    }

    // --- Check in-memory for existing tasks ---
    const allFloorTasks = this.taskManager.getFloorTasks(floorId);
    const existingInMem = allFloorTasks.filter(t => t.phaseNumber === phaseNumber);

    // --- Check Supabase for any DB-only tasks ---
    let dbCount = 0;
    try {
      dbCount = await countPhaseTasks(floorId, phaseNumber);
    } catch (err) {
      console.warn(`[EnqueueMissing] DB count failed for phase ${phaseNumber}:`, (err as Error).message);
    }

    if (existingInMem.length > 0 || dbCount > 0) {
      const taskSummary = existingInMem.map(t => ({
        taskId: t.id,
        taskType: t.taskType,
        assignedAgent: t.assignedAgent,
        status: t.status,
      }));
      console.log(
        `[EnqueueMissing] Phase ${phaseNumber} tasks already present for "${floor.name}": ` +
        `${existingInMem.length} in-memory, ${dbCount} in DB — skipping seed (idempotent)`,
      );
      await this.writeSystemReviewLog({
        event: 'enqueue_missing_phase_tasks',
        floorId, floorName: floor.name, phaseNumber,
        outcome: 'already_present',
        inMemoryCount: existingInMem.length,
        dbCount,
        enqueuedAt,
      });
      return {
        floorId, floorName: floor.name, phaseNumber,
        outcome: 'already_present',
        tasksCreated: existingInMem.length,
        taskSummary,
        enqueuedAt,
      };
    }

    // --- Tasks missing — seed them ---
    console.log(
      `[EnqueueMissing] Phase ${phaseNumber} has NO tasks for "${floor.name}" — seeding now`,
    );

    const beforeCount = allFloorTasks.length;
    try {
      await this.seedNextPhaseTasks(floorId, floor, phaseNumber - 1);
    } catch (seedErr) {
      const errMsg = (seedErr as Error).message;
      console.error(`[EnqueueMissing] seedNextPhaseTasks threw for phase ${phaseNumber}: ${errMsg}`);
      await this.writeSystemReviewLog({
        event: 'enqueue_missing_phase_tasks',
        floorId, floorName: floor.name, phaseNumber,
        outcome: 'error', error: errMsg, enqueuedAt,
      });
      return {
        floorId, floorName: floor.name, phaseNumber,
        outcome: 'error', tasksCreated: 0, taskSummary: [],
        error: errMsg, enqueuedAt,
      };
    }

    // --- Verify newly created tasks ---
    const afterTasks = this.taskManager.getFloorTasks(floorId).filter(
      t => t.phaseNumber === phaseNumber,
    );
    const tasksCreated = afterTasks.length;

    const taskSummary = afterTasks.map(t => ({
      taskId: t.id,
      taskType: t.taskType,
      assignedAgent: t.assignedAgent,
      status: t.status,
    }));

    // --- Broadcast task:created for each new task ---
    for (const t of afterTasks) {
      broadcastFloorEvent(floorId, 'task:created', {
        taskId: t.id,
        floorId,
        taskType: t.taskType,
      }).catch(() => {});
    }

    // --- Ensure PhaseManager reflects the activated phase ---
    const phases = this.phaseManager.getPhases(floorId);
    const phaseEntry = phases[phaseNumber - 1];
    if (phaseEntry && phaseEntry.status === 'pending') {
      this.phaseManager.activatePhase(floorId, phaseNumber);
    }

    // --- Persist floor.currentPhase if it's behind ---
    if (floor.currentPhase < phaseNumber) {
      floor.currentPhase = phaseNumber;
      await saveFloor(floor);
      console.log(`[EnqueueMissing] Advanced "${floor.name}" currentPhase → ${phaseNumber}`);
    }

    // --- Broadcast phase:advanced so dashboard reflects new phase ---
    await broadcastFloorEvent(floorId, 'phase:advanced', {
      from: phaseNumber - 1,
      to: phaseNumber,
      floorId,
    });

    // --- Audit ---
    await this.writeSystemReviewLog({
      event: 'enqueue_missing_phase_tasks',
      floorId, floorName: floor.name, phaseNumber,
      outcome: 'seeded', tasksCreated, taskSummary, enqueuedAt,
    });

    console.log(
      `[EnqueueMissing] ✓ Seeded ${tasksCreated} task(s) for phase ${phaseNumber} ` +
      `on "${floor.name}": ${taskSummary.map(t => t.taskType).join(', ')}`,
    );

    sendNotification({
      title: `Phase ${phaseNumber} Tasks Re-Enqueued`,
      body: `${floor.name}: ${tasksCreated} missing phase ${phaseNumber} task(s) created and queued.`,
      floorId,
      type: 'info',
    });

    return {
      floorId, floorName: floor.name, phaseNumber,
      outcome: 'seeded', tasksCreated, taskSummary, enqueuedAt,
    };
  }
