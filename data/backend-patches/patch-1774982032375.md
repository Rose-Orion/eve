  // --- Phase 9 Analytics Readiness Verification (Public API) ---

  /**
   * Query the analytics-agent Phase 9 readiness verification task status.
   * Applies business logic:
   *   - QUEUED/IN-PROGRESS → place hold on performance-review, ad-optimization, growth-report
   *   - COMPLETED+PASSED  → release held tasks in queue order
   *   - COMPLETED+FAILED  → keep hold, escalate to Floor Manager with failure details
   * Broadcasts verification results to Floor Manager in all COMPLETED cases.
   */
  async queryPhase9VerificationTask(floorId: string): Promise<{
    floorId: string;
    floorName: string;
    queriedAt: string;
    verificationStatus: 'queued' | 'in-progress' | 'completed';
    verificationResult: 'passed' | 'failed' | null;
    verificationTaskId: string | null;
    verificationDetails: string | null;
    holdApplied: boolean;
    heldTasks: Array<{ taskId: string; taskType: string; status: string }>;
    holdReason: string | null;
    tasksReleased: boolean;
    releasedTasks: Array<{ taskId: string; taskType: string; status: string }>;
    escalated: boolean;
    escalationDetails: string | null;
    broadcastSent: boolean;
    recommendation: string;
  }> {
    const queriedAt = new Date().toISOString();
    const HOLD_REASON = 'analytics-agent Phase 9 readiness verification pending';
    const PHASE9_DEPENDENT_TASK_TYPES = ['performance-review', 'ad-optimization', 'growth-report'];

    const floor = this.floors.get(floorId);
    if (!floor) {
      return {
        floorId, floorName: 'Unknown', queriedAt,
        verificationStatus: 'queued', verificationResult: null,
        verificationTaskId: null, verificationDetails: null,
        holdApplied: false, heldTasks: [], holdReason: null,
        tasksReleased: false, releasedTasks: [],
        escalated: false, escalationDetails: null, broadcastSent: false,
        recommendation: 'Floor not found — verify floor ID.',
      };
    }

    const allTasks = this.taskManager.getFloorTasks(floorId);

    // --- Locate the verification task ---
    // The readiness verification task is identified by:
    //   assignedAgent = 'analytics-agent', phaseNumber = 9,
    //   taskType contains 'readiness' OR description contains 'readiness verification'
    const verificationTask = allTasks.find(t =>
      t.assignedAgent === 'analytics-agent' &&
      t.phaseNumber === 9 &&
      (
        t.taskType.includes('readiness') ||
        t.description.toLowerCase().includes('readiness verification') ||
        t.taskType === 'analytics-readiness-check' ||
        t.taskType === 'phase9-readiness-verification'
      )
    ) ?? null;

    // Determine status
    let verificationStatus: 'queued' | 'in-progress' | 'completed';
    let verificationResult: 'passed' | 'failed' | null = null;
    let verificationDetails: string | null = null;

    if (!verificationTask) {
      // No verification task found — treat as queued (pending creation or not yet loaded)
      verificationStatus = 'queued';
      verificationDetails = 'Verification task not found in orchestrator memory. May be pending creation or in database only.';
    } else if (verificationTask.status === 'completed') {
      verificationStatus = 'completed';
      // Determine pass/fail from result content
      const result = verificationTask.result ?? '';
      const lowerResult = result.toLowerCase();
      const passPhrases = ['passed', 'ready', 'verified', 'success', 'confirmed ready', 'all checks passed'];
      const failPhrases = ['failed', 'not ready', 'unverified', 'error', 'issues found', 'checks failed'];
      const hasPassed = passPhrases.some(p => lowerResult.includes(p));
      const hasFailed = failPhrases.some(p => lowerResult.includes(p));

      if (hasFailed && !hasPassed) {
        verificationResult = 'failed';
      } else {
        // Default to passed if result exists and no explicit failure markers
        verificationResult = hasPassed || result.length > 50 ? 'passed' : null;
      }
      verificationDetails = result.slice(0, 500);
    } else if (
      verificationTask.status === 'dispatched' ||
      verificationTask.status === 'working'
    ) {
      verificationStatus = 'in-progress';
    } else {
      // created, queued, failed, retry, escalated → treat as queued (not yet confirmed)
      verificationStatus = verificationTask.status === 'escalated' ? 'completed' : 'queued';
      if (verificationTask.status === 'escalated') {
        verificationResult = 'failed';
        verificationDetails = `Verification task escalated (permanently failed): ${verificationTask.reviewFeedback ?? 'no details'}`;
      }
    }

    // --- Locate the three dependent Phase 9 tasks ---
    const dependentTasks = allTasks.filter(
      t => t.phaseNumber === 9 && PHASE9_DEPENDENT_TASK_TYPES.includes(t.taskType)
    );

    const ACTIVE_STATUSES = ['queued', 'created', 'dispatched', 'working'];

    let holdApplied = false;
    let tasksReleased = false;
    let escalated = false;
    let escalationDetails: string | null = null;
    let broadcastSent = false;
    const heldTasks: Array<{ taskId: string; taskType: string; status: string }> = [];
    const releasedTasks: Array<{ taskId: string; taskType: string; status: string }> = [];

    // --- Business Logic ---
    if (verificationStatus === 'queued' || verificationStatus === 'in-progress') {
      // RULE 1: Place hold on the three dependent tasks
      for (const task of dependentTasks) {
        if (ACTIVE_STATUSES.includes(task.status) && !task.reviewFeedback?.includes(HOLD_REASON)) {
          // Apply hold by marking reviewFeedback with hold reason
          // (does NOT change TaskStatus — preserves state machine integrity)
          task.reviewFeedback = `[HOLD] ${HOLD_REASON}`;
          task.reviewStatus = 'pending';
          try {
            const { saveTask } = await import('../integrations/supabase.js');
            await saveTask(task);
          } catch { /* non-critical */ }
          holdApplied = true;
          heldTasks.push({ taskId: task.id, taskType: task.taskType, status: task.status });
          console.log(
            `[Phase9Verification] Hold applied to task ${task.id.slice(0, 8)} ` +
            `(${task.taskType}) — reason: ${HOLD_REASON}`
          );
        } else if (task.reviewFeedback?.includes(HOLD_REASON)) {
          // Already held — include in report
          heldTasks.push({ taskId: task.id, taskType: task.taskType, status: task.status });
          holdApplied = true;
        }
      }

      // Log hold reason
      console.log(
        `[Phase9Verification] Status: ${verificationStatus}. ` +
        `Hold applied to ${heldTasks.length} task(s): ` +
        `${HOLD_REASON}`
      );

    } else if (verificationStatus === 'completed') {
      // Broadcast results to Floor Manager regardless of pass/fail
      try {
        await broadcastFloorEvent(floorId, 'phase9-verification-result', {
          floorId,
          floorName: floor.name,
          verificationResult,
          verificationDetails,
          verificationTaskId: verificationTask?.id ?? null,
          dependentTaskTypes: PHASE9_DEPENDENT_TASK_TYPES,
          queriedAt,
        });
        broadcastSent = true;
        console.log(
          `[Phase9Verification] Results broadcast to Floor Manager: ` +
          `result=${verificationResult}, floorId=${floorId}`
        );
      } catch (err) {
        console.warn(`[Phase9Verification] Broadcast failed: ${(err as Error).message}`);
      }

      if (verificationResult === 'passed') {
        // RULE 2a: Release held tasks — allow them to proceed in queue order
        for (const task of dependentTasks) {
          if (task.reviewFeedback?.includes(HOLD_REASON)) {
            // Clear the hold
            task.reviewFeedback = null;
            task.reviewStatus = 'pending';
            try {
              const { saveTask } = await import('../integrations/supabase.js');
              await saveTask(task);
            } catch { /* non-critical */ }
            releasedTasks.push({ taskId: task.id, taskType: task.taskType, status: task.status });
            console.log(
              `[Phase9Verification] Hold released for task ${task.id.slice(0, 8)} ` +
              `(${task.taskType}) — verification PASSED`
            );
          }
        }
        tasksReleased = releasedTasks.length > 0;
        console.log(
          `[Phase9Verification] PASSED — released ${releasedTasks.length} task(s). ` +
          `Tasks will proceed in queue order.`
        );

      } else if (verificationResult === 'failed') {
        // RULE 2b: Keep tasks on hold and escalate to Floor Manager
        for (const task of dependentTasks) {
          if (ACTIVE_STATUSES.includes(task.status)) {
            // Ensure hold is in place
            if (!task.reviewFeedback?.includes(HOLD_REASON)) {
              task.reviewFeedback = `[HOLD] ${HOLD_REASON} — verification FAILED, escalated`;
              task.reviewStatus = 'pending';
              try {
                const { saveTask } = await import('../integrations/supabase.js');
                await saveTask(task);
              } catch { /* non-critical */ }
            }
            heldTasks.push({ taskId: task.id, taskType: task.taskType, status: task.status });
          }
        }
        holdApplied = heldTasks.length > 0;

        escalationDetails =
          `analytics-agent Phase 9 readiness verification FAILED for "${floor.name}". ` +
          `Verification task: ${verificationTask?.id?.slice(0, 8) ?? 'unknown'}. ` +
          `Failure details: ${verificationDetails ?? 'No details available'}. ` +
          `Tasks on hold: ${PHASE9_DEPENDENT_TASK_TYPES.join(', ')}. ` +
          `These tasks will NOT execute until readiness is confirmed.`;

        // Submit escalation to Floor Manager via improvement engine
        this.improvementEngine.submitAgentFeedback(
          floorId,
          'floor-manager' as any,
          escalationDetails,
          { name: floor.name, phase: floor.currentPhase || 9, status: floor.status, goal: floor.goal },
        ).catch(() => {});

        escalated = true;
        console.error(`[Phase9Verification] FAILED — escalated to Floor Manager: ${escalationDetails}`);

        // Also send owner notification
        sendNotification({
          title: 'Phase 9 Analytics Readiness: FAILED',
          body: `analytics-agent readiness verification failed for "${floor.name}". ` +
            `Tasks performance-review, ad-optimization, growth-report are on hold pending resolution.`,
          floorId,
          type: 'alert',
        });
      }
    }

    // --- Audit record ---
    await this.writeSystemReviewLog({
      event: 'phase9_readiness_verification_query',
      floorId,
      floorName: floor.name,
      queriedAt,
      verificationTaskId: verificationTask?.id ?? null,
      verificationStatus,
      verificationResult,
      holdApplied,
      heldTaskCount: heldTasks.length,
      tasksReleased,
      releasedTaskCount: releasedTasks.length,
      escalated,
      broadcastSent,
    });

    // Build recommendation
    let recommendation: string;
    if (verificationStatus === 'queued') {
      recommendation =
        `Verification task is QUEUED (not yet dispatched). ` +
        `Hold placed on ${heldTasks.length} Phase 9 dependent task(s). ` +
        `No Phase 9 analytics tasks will execute until verification completes and passes.`;
    } else if (verificationStatus === 'in-progress') {
      recommendation =
        `Verification task is IN-PROGRESS (currently executing). ` +
        `Hold maintained on ${heldTasks.length} Phase 9 dependent task(s). ` +
        `Poll again in 30–120 seconds to check for completion.`;
    } else if (verificationResult === 'passed') {
      recommendation =
        `Verification PASSED. ${releasedTasks.length} task(s) released from hold: ` +
        `${releasedTasks.map(t => t.taskType).join(', ')}. ` +
        `Tasks will now proceed in queue order.`;
    } else if (verificationResult === 'failed') {
      recommendation =
        `Verification FAILED. Tasks on hold: ${PHASE9_DEPENDENT_TASK_TYPES.join(', ')}. ` +
        `Escalated to Floor Manager. analytics-agent must be repaired and re-verified ` +
        `before Phase 9 analytics tasks can proceed.`;
    } else {
      recommendation =
        `Verification status could not be determined. ` +
        `Check analytics-agent Phase 9 tasks manually.`;
    }

    return {
      floorId,
      floorName: floor.name,
      queriedAt,
      verificationStatus,
      verificationResult,
      verificationTaskId: verificationTask?.id ?? null,
      verificationDetails,
      holdApplied,
      heldTasks,
      holdReason: holdApplied ? HOLD_REASON : null,
      tasksReleased,
      releasedTasks,
      escalated,
      escalationDetails,
      broadcastSent,
      recommendation,
    };
  }

  // --- Phase Task Queue Investigation (Public API) ---
