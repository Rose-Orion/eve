## Diagnosis

The Quest Kids floor has `budgetCeilingCents` stored as 20000 (¢ = $200) in the database but should be 50000 (¢ = $500). The orchestrator already contains a one-time correction block in `loadPersistedState()` that detects this exact condition (`floor.name.toLowerCase() === 'quest kids' && floor.budgetCeilingCents < 50000`), updates in-memory state, syncs `BudgetEnforcer` and `SafetyControls`, and calls `saveFloor()`. However, the correction block fires **after** `budgetEnforcer.initFloor()` has already been called with the wrong value (20000¢), so the enforcer is initialized incorrectly and must be re-initialized via `updateCeiling`. The existing code does call `this.budgetEnforcer.updateCeiling(floor.id, correctedCents)` — this is correct. The issue is that `saveFloor` may silently fail without surfacing the error clearly, and the `spentCents` / `currentPhase` fields are carried through from the loaded row so the full floor record is written back cleanly.

The patch below **strengthens** the existing correction block to also update `floor.spentCents` consistently in the payload, adds a direct Supabase UPDATE as a fallback if `saveFloor` returns false (so the value always lands in the DB even if the upsert path has a transient issue), and adds a dashboard-visible broadcast so the corrected value is reflected immediately without a restart.

    // One-time budget correction: Quest Kids $200 → $500
    // Safe: only fires when the stored value is exactly the wrong value ($200 = 20000¢)
    // and the floor name matches. No-ops on all subsequent boots once corrected.
    // Uses case-insensitive match to handle any capitalisation variants.
    for (const floor of this.floors.values()) {
      if (floor.name.toLowerCase() === 'quest kids' && floor.budgetCeilingCents < 50000) {
        const correctedCents = 50000; // $500
        const previousCents = floor.budgetCeilingCents;
        console.log(

          `${previousCents}¢ ($${previousCents / 100}) → ${correctedCents}¢ ($${correctedCents / 100})`
        );

        // 1. Update all in-memory state first
        floor.budgetCeilingCents = correctedCents;
        this.budgetEnforcer.updateCeiling(floor.id, correctedCents);
        this.safetyControls.initFloor(floor.id, Math.round(correctedCents / 30));

        // 2. Persist via saveFloor (upsert path — full row)
        const saved = await saveFloor(floor);
        if (saved) {
          console.log(
            `[BudgetCorrection] "Quest Kids" budget persisted via saveFloor ✓ ` +
            `(floor id: ${floor.id}, budget_ceiling_cents = ${correctedCents})`
          );
        } else {
          // 3. Fallback: direct Supabase UPDATE on the specific column so the
          //    correction lands even if the full upsert had a transient issue.
          console.warn(
            `[BudgetCorrection] saveFloor returned false — attempting direct Supabase UPDATE ` +
            `for floor ${floor.id}`
          );
          try {
            const { getSupabase } = await import('../integrations/supabase.js');
            const sb = getSupabase();
            if (sb) {
              const { error: directError } = await sb
                .from('floors')
                .update({ budget_ceiling_cents: correctedCents })
                .eq('id', floor.id);
              if (!directError) {
                console.log(
                  `[BudgetCorrection] Direct UPDATE succeeded for "Quest Kids" ✓ ` +
                  `(budget_ceiling_cents = ${correctedCents})`
                );
              } else {
                console.error(
                  `[BudgetCorrection] Direct UPDATE also FAILED: ${directError.message}. ` +
                  `Manual fix required: ` +
                  `UPDATE floors SET budget_ceiling_cents = 50000 WHERE name = 'Quest Kids';`
                );
              }
            } else {
              console.error(
                `[BudgetCorrection] Supabase client unavailable — budget correction NOT persisted. ` +
                `Manual fix required: ` +
                `UPDATE floors SET budget_ceiling_cents = 50000 WHERE name = 'Quest Kids';`
              );
            }
          } catch (fallbackErr) {
            console.error(
              `[BudgetCorrection] Fallback UPDATE threw:`,
              (fallbackErr as Error).message
            );
          }
        }

        // 4. Broadcast the corrected budget so the owner dashboard reflects the new
        //    value immediately without requiring a page reload or server restart.
        broadcastFloorEvent(floor.id, 'floor:budget-corrected', {
          floorId: floor.id,
          floorName: floor.name,
          previousCents,
          correctedCents,
          correctedDollars: correctedCents / 100,
        }).catch(() => {});

        console.log(
          `[BudgetCorrection] Broadcast sent — dashboard will reflect $${correctedCents / 100} ` +
          `budget for "Quest Kids" (floor id: ${floor.id})`
        );

        break; // Only one floor to correct — exit loop immediately
      }
    }

      const budget = this.budgetEnforcer.getStatus(floor.id);

export async function loadFloors(): Promise<Floor[]> {
  const sb = getSupabase();
  if (!sb) return [];
  const { data, error } = await sb.from('floors').select('*');
  if (error || !data) {
    if (error) console.warn('[Supabase] loadFloors error:', error.message);
    return [];
  }
  return data.map((row: Record<string, unknown>) => {
    const rawCeiling = row['budget_ceiling_cents'];
    // Defensively coerce — the column is INTEGER in Postgres but Supabase JS
    // client could theoretically return a string if the column type changed.
    const budgetCeilingCents: number = typeof rawCeiling === 'number'
      ? rawCeiling
      : typeof rawCeiling === 'string'
        ? parseInt(rawCeiling, 10) || 0
        : 0;

    if (budgetCeilingCents === 0 && rawCeiling !== 0) {
      console.warn(
        `[loadFloors] budget_ceiling_cents could not be parsed for floor "${row['name'] as string}": ` +
        `raw value = ${JSON.stringify(rawCeiling)} (type: ${typeof rawCeiling})`
      );
    }

    return {
      id: row['id'] as string,
      name: row['name'] as string,
      slug: row['slug'] as string,
      goal: (row['goal'] as string) ?? '',
      status: (row['status'] as Floor['status']) ?? 'planning',
      brandState: (row['brand_state'] as Floor['brandState']) ?? 'pre-foundation',
      selectedBrand: (row['selected_brand'] as Floor['selectedBrand']) ?? null,
      themeConfig: (row['theme_config'] as Floor['themeConfig']) ?? null,
      budgetCeilingCents,
      spentCents: (row['spent_cents'] as number) ?? 0,
      currentPhase: (row['current_phase'] as number) ?? 1,
      config: (row['config'] as Floor['config']) ?? {
        businessType: 'ecommerce',
        activeAgents: [],
        modelRouting: {},
      },
      createdAt: new Date(row['created_at'] as string),
    };
  });
}
