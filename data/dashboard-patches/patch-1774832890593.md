## Diagnosis

The current `viewChat` function renders the Floor Manager as a **full-page view** that replaces the floor content entirely. The CSS file already has complete chat panel styles defined (`.chat-panel-*`, `#floor-chat-panel`, etc.), but the JavaScript never creates the panel DOM element, never reads/writes the panel state from localStorage, and the "💬 Chat" button in `viewFloorBuilding` still routes to `'floor-chat'` (a full page). The fix requires: (1) replacing `viewChat` and `sendChat` with a persistent panel implementation, (2) adding `initChatPanel()` / `openChatPanel()` functions, (3) updating the "💬 Chat" buttons to open the panel instead of navigating, (4) calling `initChatPanel()` in `init()`, and (5) exposing the new globals.

// Panel state persisted in localStorage
const CHAT_PANEL_KEY = 'eve-chat-panel-state';

function getChatPanelState() {
  try {
    return JSON.parse(localStorage.getItem(CHAT_PANEL_KEY) || '{}');
  } catch { return {}; }
}

function saveChatPanelState(patch) {
  const s = getChatPanelState();
  localStorage.setItem(CHAT_PANEL_KEY, JSON.stringify({ ...s, ...patch }));
}

// Per-floor message store in state
// state.chatHistories = { floorId: [ {role, text, ts} ] }
if (!state.chatHistories) state.chatHistories = {};

function getChatHistory(floorId) {
  if (!state.chatHistories[floorId]) {
    const floor = state.floors.find(f => f.id === floorId);
    const name = floor?.name || 'your business';
    state.chatHistories[floorId] = [
      { role: 'assistant', text: `Hi! I'm the Floor Manager for ${name}. Ask me anything about your business, strategy, or progress.`, ts: Date.now() },
    ];
  }
  return state.chatHistories[floorId];
}

function formatBubbleTime(ts) {
  if (!ts) return '';
  const d = new Date(ts);
  return d.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
}

function renderChatPanelMessages(floorId) {
  const panel = document.getElementById('floor-chat-panel');
  if (!panel) return;
  const msgs = document.getElementById('chat-panel-messages');
  if (!msgs) return;

  const history = getChatHistory(floorId);

  if (history.length === 0) {
    msgs.innerHTML = `
      <div class="chat-panel-empty">
        <div class="chat-panel-empty-icon">🤖</div>
        <div>Ask the Floor Manager anything about your business.</div>
      </div>`;
    return;
  }

  msgs.innerHTML = history.map(m => `
    <div class="chat-bubble-row ${m.role}">
      <div class="chat-bubble">${m.text.replace(/\n/g, '<br>')}</div>
      ${m.ts ? `<div class="chat-bubble-time">${formatBubbleTime(m.ts)}</div>` : ''}
    </div>
  `).join('');

  // Scroll to bottom
  requestAnimationFrame(() => { msgs.scrollTop = msgs.scrollHeight; });
}

function applyPanelVisualState() {
  const panel = document.getElementById('floor-chat-panel');
  if (!panel) return;
  const s = getChatPanelState();
  const toggleBtn = document.getElementById('chat-panel-toggle-btn');

  if (!s.open) {
    // Fully hidden — no class on body, panel stays translateX(100%)
    panel.classList.remove('panel-expanded', 'panel-minimized');
    document.body.classList.remove('chat-panel-open', 'chat-panel-minimized');
    return;
  }

  if (s.minimized) {
    panel.classList.remove('panel-expanded');
    panel.classList.add('panel-minimized');
    document.body.classList.remove('chat-panel-open');
    document.body.classList.add('chat-panel-minimized');
    if (toggleBtn) toggleBtn.textContent = '▶';
  } else {
    panel.classList.remove('panel-minimized');
    panel.classList.add('panel-expanded');
    document.body.classList.remove('chat-panel-minimized');
    document.body.classList.add('chat-panel-open');
    if (toggleBtn) toggleBtn.textContent = '◀';
  }
}

function openChatPanel(floorId) {
  buildChatPanelDOM();
  saveChatPanelState({ open: true, floorId: floorId || null });

  // Restore minimized preference, default to expanded
  const s = getChatPanelState();
  if (s.minimized === undefined) saveChatPanelState({ minimized: false });

  applyPanelVisualState();

  // Update floor info in header
  const floor = floorId ? state.floors.find(f => f.id === floorId) : null;
  const nameEl = document.getElementById('chat-panel-floor-name');
  const subEl  = document.getElementById('chat-panel-floor-sub');
  if (nameEl) nameEl.textContent = floor ? `Floor Manager — ${floor.name}` : 'Floor Manager';
  if (subEl)  subEl.textContent  = floor ? (floor.goal || 'Ask anything about your business') : 'Select a floor to chat';

  // Render messages
  if (floorId) renderChatPanelMessages(floorId);

  // Focus input
  requestAnimationFrame(() => {
    const input = document.getElementById('chat-panel-input');
    if (input && !getChatPanelState().minimized) input.focus();
  });
}

function closeChatPanel() {
  saveChatPanelState({ open: false });
  const panel = document.getElementById('floor-chat-panel');
  if (panel) {
    panel.classList.remove('panel-expanded', 'panel-minimized');
    document.body.classList.remove('chat-panel-open', 'chat-panel-minimized');
  }
}

function toggleChatPanel() {
  const s = getChatPanelState();
  saveChatPanelState({ minimized: !s.minimized });
  applyPanelVisualState();

  // If expanding, re-focus input and scroll messages
  if (!getChatPanelState().minimized) {
    requestAnimationFrame(() => {
      const input = document.getElementById('chat-panel-input');
      if (input) input.focus();
      const msgs = document.getElementById('chat-panel-messages');
      if (msgs) msgs.scrollTop = msgs.scrollHeight;
    });
  }
}

function chatPanelKeydown(e) {
  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault();
    sendChatPanelMessage();
  }
}

async function sendChatPanelMessage() {
  const input = document.getElementById('chat-panel-input');
  if (!input || !input.value.trim()) return;

  const msg = input.value.trim();
  input.value = '';
  input.style.height = 'auto';

  const floorId = getChatPanelState().floorId;
  if (!floorId) {
    toast('Open a floor first to chat with the Floor Manager.', 'info');
    return;
  }

  const history = getChatHistory(floorId);
  history.push({ role: 'user', text: msg, ts: Date.now() });
  renderChatPanelMessages(floorId);

  // Show typing indicator
  const msgs = document.getElementById('chat-panel-messages');
  if (msgs) {
    const typing = document.createElement('div');
    typing.className = 'chat-bubble-row assistant';
    typing.id = 'chat-typing-indicator';
    typing.innerHTML = `
      <div class="chat-typing">
        <div class="chat-typing-dot"></div>
        <div class="chat-typing-dot"></div>
        <div class="chat-typing-dot"></div>
      </div>`;
    msgs.appendChild(typing);
    msgs.scrollTop = msgs.scrollHeight;
  }

  const res = await api('POST', `/api/chat/${floorId}/message`, { message: msg });

  // Remove typing indicator
  document.getElementById('chat-typing-indicator')?.remove();

  history.push({
    role: 'assistant',
    text: res?.response || res?.reply || 'Floor Manager is temporarily unavailable.',
    ts: Date.now(),
  });
  renderChatPanelMessages(floorId);
}

function initChatPanel() {
  buildChatPanelDOM();
  const s = getChatPanelState();
  if (s.open) {
    applyPanelVisualState();
    if (s.floorId) {
      const floor = state.floors.find(f => f.id === s.floorId);
      const nameEl = document.getElementById('chat-panel-floor-name');
      const subEl  = document.getElementById('chat-panel-floor-sub');
      if (nameEl) nameEl.textContent = floor ? `Floor Manager — ${floor.name}` : 'Floor Manager';
      if (subEl)  subEl.textContent  = floor ? (floor.goal || 'Ask anything about your business') : 'Select a floor to chat';
      renderChatPanelMessages(s.floorId);
    }
  }
}
